CREATE DATABASE DB_AGENDA;

USE DB_AGENDA;

CREATE TABLE TB_USUARIOS (
USU_CODIGO INT AUTO_INCREMENT PRIMARY KEY,
USU_NOME VARCHAR(50) NOT NULL,
USU_EMAIL VARCHAR(50) NOT NULL,
USU_SENHA VARCHAR(50) NOT NULL
) ENGINE=InnoDB;

CREATE TABLE TB_AGENDAS (
AGE_CODIGO INT AUTO_INCREMENT PRIMARY KEY,
AGE_NOME VARCHAR (50) NOT NULL,
AGE_USU_CODIGO INT REFERENCES TB_USUARIOS(USU_CODIGO)
) ENGINE=InnoDB;

CREATE TABLE TB_EVENTOS(
EVE_CODIGO INT AUTO_INCREMENT PRIMARY KEY,
EVE_TITULO VARCHAR(50) NOT NULL, 
EVE_DESCRICAO VARCHAR(50) NOT NULL, 
EVE_DATAINICIO TIMESTAMP NOT NULL,
EVE_DATAFIM TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
EVE_AGE_CODIGO INT REFERENCES TB_AGENDAS(AGE_CODIGO)
) ENGINE=InnoDB;
